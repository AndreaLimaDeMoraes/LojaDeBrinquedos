-- Criar banco
CREATE DATABASE LojaBrinquedo;

-- Usar o banco
USE LojaBrinquedo;

-- Tabela usuarios
CREATE TABLE usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL
);

-- Tabela marcas
CREATE TABLE marcas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
    logo VARCHAR(255)
);

-- Tabela categorias
CREATE TABLE categorias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    icone VARCHAR(255)
);

-- Tabela brinquedos
CREATE TABLE brinquedos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    imagens TEXT,
    valor DECIMAL(10,2) NOT NULL,
    descricao TEXT,
    fk_categoria INT,
    fk_usuario INT,
    idade_recomendada VARCHAR(50),
    quantidade_estoque INT DEFAULT 0,
    desconto DECIMAL(5,2) DEFAULT 0,
    data_cadastro DATETIME DEFAULT CURRENT_TIMESTAMP,
    detalhes TEXT,
    fk_marca INT,

    FOREIGN KEY (fk_categoria) REFERENCES categorias(id),
    FOREIGN KEY (fk_usuario) REFERENCES usuarios(id),
    FOREIGN KEY (fk_marca) REFERENCES marcas(id)
);

-- Tabela de relacionamento brinquedos_categorias
CREATE TABLE brinquedos_categorias (
    fk_brinquedo INT,
    fk_categoria INT,

    PRIMARY KEY (fk_brinquedo, fk_categoria),

    FOREIGN KEY (fk_brinquedo) REFERENCES brinquedos(id),
    FOREIGN KEY (fk_categoria) REFERENCES categorias(id)
);

-- Tabela informacoes do site
CREATE TABLE informacoes_site (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome_loja VARCHAR(150),
    logo VARCHAR(255),
    infos TEXT
);
